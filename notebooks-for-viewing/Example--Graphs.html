def lps(objs):
    return " ".join(o if type(o) is str else sp.latex(o) for o in objs)

def lpl(objs):
    return "$" + lps(objs) + "$"

def lp(objs):
    display(ipd.Math(lps(objs)))

def px(obj, d, f, *args, **kwargs):
    result = getattr(obj, f)(*args, **kwargs)
    if d or f in deco:
        if not d:
            d = deco[f]
        display(ipd.Math(d.format(o=sp.latex(obj), a=args, **kwargs) + ' = ' + sp.latex(result)))
    else:
        lp([obj, '=', result])
    return result

def pxn(objl, objr, d, f, *args, **kwargs):
    resultr = getattr(objr, f)(*args, **kwargs)
    if d or f in deco:

    else:
        lp([objl, '=', objr, '=', resultr])
    return resultr

import matplotlib.ticker as tck

def multiple_formatter(denominator=2, number=np.pi, latex='\pi'):
    def gcd(a, b):
        while b:
            a, b = b, a%b
        return a
    def _multiple_formatter(x, pos):
        den = denominator
        num = np.int(np.rint(den*x/number))
        com = gcd(num,den)
        (num,den) = (int(num/com),int(den/com))
        if den==1:
            if num==0:
                return r'$0$'
            if num==1:
                return r'$%s$'%latex
            elif num==-1:
                return r'$-%s$'%latex
            else:
                return r'$%s%s$'%(num,latex)
        else:
            if num==1:
                return r'$\frac{%s}{%s}$'%(latex,den)
            elif num==-1:
                return r'$\frac{-%s}{%s}$'%(latex,den)
            else:
                return r'$\frac{%s%s}{%s}$'%(num,latex,den)
    return _multiple_formatter

class Multiple:
    def __init__(self, denominator=2, number=np.pi, latex='\pi'):
        self.denominator = denominator
        self.number = number
        self.latex = latex

    def locator(self):
        return plt.MultipleLocator(self.number / self.denominator)

    def formatter(self):
        return plt.FuncFormatter(multiple_formatter(self.denominator, self.number, self.latex))
